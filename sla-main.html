function updateKpiBar(jobs){ let total = jobs.length; let open = 0, overdue = 0, dueToday = 0, atRisk = 0; const today = new Date(); today.setHours(0, 0, 0, 0); const endToday = new Date(today); endToday.setHours(23, 59, 59, 999); for(const job of jobs){ const status = (job.status || "").toLowerCase(); if(status === "open" || status === "in progress" || status === "on hold") open++; const sla = computeSlaCountdown(job.targetAt || job.sla?.targetAt); if(sla.status === "overdue") overdue++; if(sla.status === "warn" || sla.status === "overdueSoon") atRisk++; const targetDate = parseDate(job.targetAt || job.sla?.targetAt); if(targetDate && targetDate >= today && targetDate <= endToday) dueToday++; } $("kpiTotal").textContent = total; $("kpiOpen").textContent = open; $("kpiOverdue").textContent = overdue; $("kpiToday").textContent = dueToday; $("kpiRisk").textContent = atRisk; }
